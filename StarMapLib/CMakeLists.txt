project( StarMapLib )

	${CMAKE_SOURCE_DIR}
	${CMAKE_CURRENT_SOURCE_DIR}
)

add_target_precompiled_header( ${PROJECT_NAME} pch.h pch.cpp ${CMAKE_CXX_FLAGS} ${${PROJECT_NAME}_SRC_FILES} )
add_dependencies( ${PROJECT_NAME}
	GlLib
	RenderLib
)
